CREATE TABLE Color (
	color_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	colorName VARCHAR(20)
);

CREATE TABLE Texture (
	texture_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	textureName VARCHAR(20)
);

CREATE TABLE Variant (
	variant_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	variantName VARCHAR(50)
);

CREATE TABLE Cardboard (
	cardboard_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	cardboardName VARCHAR(20),	
	sampleQuantity INT,
	bagQuantity INT,
	boxQuantity INT
);

CREATE TABLE Palette (
	palette_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	paletteName VARCHAR(20),
	planeQuantity INT,
	boatQuantity INT,
	truckQuantity INT,
	fk_cardboard_id INT,
	FOREIGN KEY (fk_cardboard_id) REFERENCES Cardboard(cardboard_id)
);

CREATE TABLE Packaging (
	packaging_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	packagingName VARCHAR(20),
	content INT,
	quantity INT,
	fk_palette_id INT,
	FOREIGN KEY (fk_palette_id) REFERENCES Palette(palette_id)
);

CREATE TABLE Country (
	country_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	countryName VARCHAR(50),
	fk_packaging_id INT,
	FOREIGN KEY (fk_packaging_id) REFERENCES Packaging(packaging_id)
);


CREATE TABLE Machine (
	machine_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	machineNumber INT,
	cadence INT,
	machineDelay INT,
	fk_variant_id INT,
	fk_packaging_id INT,
	FOREIGN KEY (fk_variant_id) REFERENCES Variant(variant_id),
	FOREIGN KEY (fk_packaging_id) REFERENCES Packaging(packaging_id)
);

CREATE TABLE Candy (
	candy_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	candyName VARCHAR(30),
	additive INT,
	coating INT,
	aroma INT,
	gelling INT,
	sugar INT
);

CREATE TABLE CandyCost (
	candyCost_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	manufacturePercentage FLOAT,
	contioningPercentage FLOAT,
	shippingPercentage FLOAT,
	generalPercentage FLOAT,
	sampleCost FLOAT,
	bagCost FLOAT,
	boxCost FLOAT,
	fk_candy_id INT,
	FOREIGN KEY (fk_candy_id) REFERENCES Candy(candy_id)
);

CREATE TABLE CandyReferences (
	candyReference_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	candyStock INT NOT NULL,
	fk_candy_id INT,
	fk_color_id INT,
	fk_texture_id INT,
	fk_variant_id INT, 
	fk_packaging_id INT,
	FOREIGN KEY (fk_candy_id) REFERENCES Candy(candy_id),
	FOREIGN KEY (fk_color_id) REFERENCES Color(color_id),
	FOREIGN KEY (fk_texture_id) REFERENCES Texture(texture_id),
	FOREIGN KEY (fk_variant_id) REFERENCES Variant(variant_id),
	FOREIGN KEY (fk_packaging_id) REFERENCES Packaging(packaging_id)
);

CREATE TABLE Orders (
	order_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	orderDate DATE,
	clientName VARCHAR(50),
	clientSurname VARCHAR(50),
  clientMail VARCHAR(50),
	quantity INT,
	totalCost FLOAT,
  orderState VARCHAR(20),
  trackingNumber INT,
	fk_candyReference_id INT,
	fk_country_id INT,
	FOREIGN KEY (fk_candyReference_id) REFERENCES CandyReferences(candyReference_id),
	FOREIGN KEY (fk_country_id) REFERENCES Country(country_id)
);

CREATE TABLE Stock (
	stock_id NUMBER GENERATED BY DEFAULT ON NULL AS IDENTITY PRIMARY KEY,
	component VARCHAR(10),
	conditionnementKg INT,
	paletteKg INT
);
